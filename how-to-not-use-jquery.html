let + const

// Doesn't work as expected
for (var i = 0; i < 10; i++) {
	setTimeout(function () {
		console.log(i);
	}, 100);
}

// Works as expected
for (var i = 0; i < 10; i++) {
	let j = i;
	setTimeout(function () {
		console.log(j);
	}, 100);
}
